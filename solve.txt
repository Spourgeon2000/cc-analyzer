    function getDtTimeZone() {
        var rightNow = new Date();
        var toDay = new Date(rightNow.getFullYear(), 0, 1, 0, 0, 0, 0);
        var temp = toDay.toGMTString();
        var date2 = new Date(temp.substring(0, temp.lastIndexOf(" ") - 1));
